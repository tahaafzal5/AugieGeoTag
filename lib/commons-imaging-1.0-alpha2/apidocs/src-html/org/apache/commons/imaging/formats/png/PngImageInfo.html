017package org.apache.commons.imaging.formats.png;
018
019import java.util.ArrayList;
020import java.util.List;
021
022import org.apache.commons.imaging.ImageFormat;
023import org.apache.commons.imaging.ImageInfo;
024
025public class PngImageInfo extends ImageInfo {
026    private final List<PngText> textChunks;
027    private final PhysicalScale physicalScale;
028
029    PngImageInfo(final String formatDetails, final int bitsPerPixel,
030            final List<String> comments, final ImageFormat format, final String formatName,
031            final int height, final String mimeType, final int numberOfImages,
032            final int physicalHeightDpi, final float physicalHeightInch,
033            final int physicalWidthDpi, final float physicalWidthInch, final int width,
034            final boolean progressive, final boolean transparent, final boolean usesPalette,
035            final ColorType colorType, final CompressionAlgorithm compressionAlgorithm,
036            final List<PngText> textChunks, final PhysicalScale physicalScale) {
037        super(formatDetails, bitsPerPixel, comments, format, formatName,
038                height, mimeType, numberOfImages, physicalHeightDpi,
039                physicalHeightInch, physicalWidthDpi, physicalWidthInch, width,
040                progressive, transparent, usesPalette, colorType,
041                compressionAlgorithm);
042
043        this.textChunks = textChunks;
044        this.physicalScale = physicalScale;
045    }
046
047    public List<PngText> getTextChunks() {
048        return new ArrayList<>(textChunks);
049    }
050
051   /**
052    * Physical scale of Image.
053    *
054    * @return {@link PhysicalScale}. If undefined then {@link PhysicalScale#UNDEFINED}
055    * is returned.
056    */
057    public PhysicalScale getPhysicalScale() {
058        return physicalScale;
059    }
060}
