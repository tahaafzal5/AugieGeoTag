017package org.apache.commons.imaging.formats.tiff.fieldtypes;
018
019import java.nio.ByteOrder;
020
021import org.apache.commons.imaging.ImageWriteException;
022import org.apache.commons.imaging.common.ByteConversions;
023import org.apache.commons.imaging.common.RationalNumber;
024import org.apache.commons.imaging.formats.tiff.TiffField;
025
026public class FieldTypeRational extends FieldType {
027    public FieldTypeRational(final int type, final String name) {
028        super(type, name, 8);
029    }
030
031    @Override
032    public Object getValue(final TiffField entry) {
033        final byte[] bytes = entry.getByteArrayValue();
034        if (entry.getCount() == 1) {
035            return ByteConversions.toRational(bytes,
036                    entry.getByteOrder());
037        }
038        return ByteConversions.toRationals(bytes, entry.getByteOrder());
039    }
040
041    @Override
042    public byte[] writeData(final Object o, final ByteOrder byteOrder) throws ImageWriteException {
043        if (o instanceof RationalNumber) {
044            return ByteConversions.toBytes((RationalNumber) o, byteOrder);
045        } else if (o instanceof RationalNumber[]) {
046            return ByteConversions.toBytes((RationalNumber[]) o, byteOrder);
047        } else if (o instanceof Number) {
048            final Number number = (Number) o;
049            final RationalNumber rationalNumber = RationalNumber.valueOf(number.doubleValue());
050            return ByteConversions.toBytes(rationalNumber, byteOrder);
051        } else if (o instanceof Number[]) {
052            final Number[] numbers = (Number[]) o;
053            final RationalNumber[] rationalNumbers = new RationalNumber[numbers.length];
054            for (int i = 0; i < numbers.length; i++) {
055                final Number number = numbers[i];
056                rationalNumbers[i] = RationalNumber.valueOf(number.doubleValue());
057            }
058            return ByteConversions.toBytes(rationalNumbers, byteOrder);
059        } else if (o instanceof double[]) {
060            final double[] numbers = (double[]) o;
061            final RationalNumber[] rationalNumbers = new RationalNumber[numbers.length];
062            for (int i = 0; i < numbers.length; i++) {
063                final double number = numbers[i];
064                rationalNumbers[i] = RationalNumber.valueOf(number);
065            }
066            return ByteConversions.toBytes(rationalNumbers, byteOrder);
067        } else {
068            throw new ImageWriteException("Invalid data", o);
069        }
070    }
071}
